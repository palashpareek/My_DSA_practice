class Solution {
public:
    int findDuplicate(vector<int>& nums) {
        
        int n = nums.size();
        
       int tortoise = nums[0];
       int hare =  nums[0];
        
        do{ 
            
          tortoise = nums[tortoise];
            hare = nums[nums[hare]];
}while (tortoise!=hare);
            
        hare= nums[0];
            while(tortoise!=hare)
            {
                 tortoise = nums[tortoise];
            hare = nums[hare];
}
        return tortoise; 
        
    }
};
